```
library(tidyverse)
library(zoo)

data %>%
  mutate(seven_avg= rollmean(<target_column>, 7, align="left", fill=0)) %>%
  ggplot(aes(x = ...) +
  geom_line(aes(y = seven_avg), color = "red", size = .75)
```
